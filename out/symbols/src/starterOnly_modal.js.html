 59 function addValidation(input) {
 60   switch (input.type) {
 61     case "text":
 62       input.oninput = function () {
 63         validText(input);
 64       };
 65       break;
 66     case "email":
 67       input.oninput = function () {
 68         validMail(input);
 69       };
 70       break;
 71     case "date":
 72       limitDate(input, "max", 18); // Age minimun 18ans
 73       limitDate(input, "min", 100); // Age maximum 100ans
 74       input.onchange = function () {
 75         validDate(input);
 76       };
 77       break;
 78     case "number":
 79       input.oninput = function () {
 80         validNumber(input, document.querySelector("input[name = location]"));
 81       };
 82       break;
 83     case "checkbox":
 84       if (input.name === "location") {
 85         input.onchange = function () {
 86           // @ts-ignore
 87           validCheckboxLocation(input, quantity.value, quantity);
 88         };
 89       } else {
 90         input.onchange = function () {
 91           validCheckboxConditions(input, checkbox1);
 92         };
 93       }
 94       break;
 95     case "submit":
 96       input.onclick = function (e) {
 97         e.preventDefault();
 98         if (errorTest(formDatas)) {
 99           showMessage(input, "Merci ! Votre réservation a été reçue.");
100           console.log("Merci ! Votre réservation a été reçue.");
101         } else {
102           inputs.forEach((input) => {
103             switch (input.type) {
104               case "text":
105                 validText(input);
106                 break;
107               case "email":
108                 validMail(input);
109                 break;
110               case "date":
111                 validDate(input);
112                 break;
113               case "number":
114                 validNumber(
115                   input,
116                   document.querySelector("input[name = location]")
117                 );
118                 break;
119               case "checkbox":
120                 if (input.name === "location") {
121                   validCheckboxLocation(input, quantity.value, quantity);
122                 } else {
123                   validCheckboxConditions(input, checkbox1);
124                 }
125                 break;
126             }
127           });
128         }
129       };
130       break;
131     default:
132       break;
133   }
134 }

166 /**
167  * Validation input[type="text"]
168  *
169  * @param   {HTMLInputElement}  input  input[type="text"]

171  * @return    {void}                      Message error

174 function validText(input) {
175   let value = input.value.trim();
176   if (value.length < 2)
177     return showMessage(input, "Veuillez entrez au moins 2 caractères");
178   // si pas regex return showMessage...
179   else if (!firstLastRegex.test(value))
180     return showMessage(
181       input,
182       "Veuillez entrez seulement des caractéres litterales"
183     );
184   deleteMessage(input);

189 /**
190  * Ajoute un zero devant date si date < 10
191  * @param {string | number} dateNumber jour ou mois
192  * @returns {string} 0+jours ou 0+mois
193  */
194 function addZero(dateNumber) {
195   if (dateNumber < 10) return "0" + dateNumber;
196   return dateNumber.toString();
197 }

199 /**
200  * donne une limite à l'input
201  * @param {HTMLInputElement} input input du formulaire
202  * @param {("min" | "max")} type l'extrémité à limiter
203  * @param {Number} gap le nombre d'années à décaler
204  * @returns {void} met à jour l'input avec une limite
205  */
206 function limitDate(input, type, gap) {
207   const now = new Date(); // Récupération de la date actuelle
208   const year = now.getFullYear() - gap; // Année actuelle -18ans ou -100ans
209   const month = addZero(now.getUTCMonth()); // Ajout du zero(Mois actuel)
210   const day = addZero(now.getUTCDate()); // Ajout du zero(jour actuel)
211   input.setAttribute(type, `${year}-${month}-${day}`); // Ajoute l'attribut "Min" ou "max" avec la date correspondante.
212 }

214 /**
215  * Validation input[type="date"]
216  *
217  * @param {HTMLInputElement} input  input du formulaire
218  * @returns {Void} Affiche ou supprime le message d'erreur
219  */
220 function validDate(input) {
221   let value = input.value;
222   let min = input.min;
223   let max = input.max;
224   if (value < min) return showMessage(input, "Age maximum 100 ans");
225   else if (value > max) return showMessage(input, "Age minimum 18 ans");
226   deleteMessage(input);
227 }

245 /**
246  * @param {HTMLInputElement} input input du formulaire
247  * @param {HTMLInputElement} cible input du formulaire qui recevra le message erreur
248  */
249 function validNumber(input, cible) {
250   let value = input.value;
251   if (parseInt(value) < 0 || parseInt(value) > 100 || value == "") {
252     deleteMessage(cible);
253     return showMessage(input, "Veuillez entrez une valeur entre 0 et 100");
254   } else if (parseInt(value) > 0 && numberOfLocationChecked() === 0) {
255     deleteMessage(input);
256     return showMessage(cible, "Veuillez selectionner une ville");
257   } else if (parseInt(value) == 0 && numberOfLocationChecked() > 0) {
258     deleteMessage(input);
259     return showMessage(
260       cible,
261       "Vous ne pouvez pas selectionner une ville si vous n'avez jamais participé à un tournoi"
262     );
263   } else if (parseInt(value) < numberOfLocationChecked()) {
264     return showMessage(
265       cible,
266       "Vous ne pouvez pas selectionner plus de villes que de tournoi participé"
267     );
268   } else if (value == "" && numberOfLocationChecked() === 0) {
269     deleteMessage(cible);
270     return showMessage(input, "Veuillez entrez une valeur entre 0 et 100");
271   } else if (parseInt(value) > 0 && numberOfLocationChecked() > 0) {
272     deleteMessage(cible);
273   }
274 
275   deleteMessage(input);
276 }

294       input,

296     );
297   }

302 function numberOfLocationChecked() {
303   const inputsLocations = document.querySelectorAll("input[name = location]");
304   let numberInputLocationChecked = 0;
305   for (let inputLocation of inputsLocations) {
306     // @ts-ignore
307     if (inputLocation.checked) {
308       numberInputLocationChecked++;
309     }
310   }
311   return numberInputLocationChecked;

333 /**
